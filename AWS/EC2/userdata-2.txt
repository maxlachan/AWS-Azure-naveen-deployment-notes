# Use this for your user data (script from top to bottom)
#!/bin/bash

exec > /var/log/user-data.log 2>&1
set -x

yum update -y

# Install Node.js
curl -fsSL https://rpm.nodesource.com/setup_22.x | bash -
yum install -y nodejs

# Install Git
yum install -y git

# Install PM2
npm install -g pm2

# Clone repo
cd /home/ec2-user
git clone https://github.com/naveen-uppala/frontend.git
cd frontend || exit

# Install dependencies & build frontend
npm install
npm run build

# Move to server
cp -r build server
cd server || exit

npm install

# Start app with PM2
pm2 start index.js
pm2 save
pm2 startup
